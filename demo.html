$(document).ready(function() { $(".hide-show").on("click", function(e) { e.preventDefault(); //open and close menu item $(this) .next(".content") .slideToggle(1000); //close menu item if another one opened $(this) .siblings() .next(".content") .slideUp(1000); //keep menu on screen if expanding downward $(".insideWrapper").scrollTop($(".insideWrapper")[0].scrollHeight); }); /*THIS ADDS CLICK OUTSIDE BOX [INSIDEWRAPPER] TO CLOSE*/ $("body").mouseup(function(e) { var subject = $(".insideWrapper"); var content = $(".content"); if (e.target.id != subject.attr("id") && !subject.has(e.target).length) { content.slideUp(1000); } }); });
